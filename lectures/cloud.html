  req: NextApiRequest,
  res: NextApiResponse
) {

    const users = await db.user.findMany();
    return res.json(users);
  }

Heavy Compute

Video processing, ML inference, data pipelines won't work well.

Use: AWS/GCP for compute
🗄️

Database Hosting

Vercel doesn't host databases. Need external provider.

Use: Supabase, PlanetScale, Neon

Cost at Scale

Can get expensive with high traffic. Bandwidth costs add up.

Consider: Self-hosting for very high traffic
Vercel is perfect for: Frontend apps, marketing sites, blogs, dashboards, APIs with moderate traffic.

Supabase Database Queries

// Create (Insert)
const { data, error } = await supabase
  .from('posts')
  .insert({ title: 'Hello World', author_id: user.id })
  .select()

// Read (Select with filters)
const { data: posts } = await supabase
  .from('posts')
  .select('*, author:users(name, avatar)')
  .eq('published', true)
  .order('created_at', { ascending: false })
  .limit(10)

// Update
const { data } = await supabase
  .from('posts')
  .update({ title: 'Updated Title' })
  .eq('id', postId)
  .select()

// Delete
await supabase
  .from('posts')
  .delete()
  .eq('id', postId)

// Realtime subscription
supabase
  .channel('posts')
  .on('postgres_changes', { event: '*', schema: 'public', table: 'posts' },
    (payload) => console.log('Change:', payload)
  )
  .subscribe()

Row Level Security

Users can only see their own data

Row Level Security (RLS)

Secure your data at the database level

-- Enable RLS on a table
ALTER TABLE posts ENABLE ROW LEVEL SECURITY;

-- Policy: Users can only see their own posts
CREATE POLICY "Users can view own posts"
ON posts FOR SELECT
USING (auth.uid() = author_id);

-- Policy: Users can only insert their own posts
CREATE POLICY "Users can create own posts"
ON posts FOR INSERT
WITH CHECK (auth.uid() = author_id);

-- Policy: Users can update their own posts
CREATE POLICY "Users can update own posts"
ON posts FOR UPDATE
USING (auth.uid() = author_id);

-- Policy: Public posts are visible to everyone
CREATE POLICY "Public posts are visible"
ON posts FOR SELECT
USING (published = true);

-- Now even if someone bypasses your frontend,
-- the DATABASE enforces these rules!
Why RLS matters: Your API can be simple because security is enforced at the data layer, not in every endpoint.

When NOT to Use Supabase

Massive Scale

Single Postgres instance has limits. Complex sharding not built-in.

Use: CockroachDB, Spanner for global scale
🔧

Complex Backend Logic

Edge functions have limits. No background jobs, limited runtime.

Use: Railway, Render for custom backends
📄

Document-Heavy Workloads

It's Postgres. If you need MongoDB-style flexibility, look elsewhere.

Use: MongoDB Atlas, Fauna
🔒

Vendor Lock-in Concerns

It's open source! You can self-host if needed.

Self-host: supabase.com/docs/self-hosting

Full Stack Example: Todo App

// app/page.tsx - Next.js + Supabase
'use client'
import { useEffect, useState } from 'react'
import { createClient } from '@/lib/supabase/client'

export default function Todos() {
  const [todos, setTodos] = useState([])
  const supabase = createClient()

  useEffect(() => {
    // Fetch initial todos
    supabase.from('todos').select('*').then(({ data }) => setTodos(data || []))

    // Subscribe to realtime changes
    const channel = supabase
      .channel('todos')
      .on('postgres_changes', { event: '*', schema: 'public', table: 'todos' },
        (payload) => {
          if (payload.eventType === 'INSERT') {
            setTodos(prev => [...prev, payload.new])
          }
        }
      )
      .subscribe()

    return () => { supabase.removeChannel(channel) }
  }, [])

  const addTodo = async (text: string) => {
    const { data: { user } } = await supabase.auth.getUser()
    await supabase.from('todos').insert({ text, user_id: user?.id })
  }

  return (/* UI */)
}

Environment Variables & Secrets

// NEVER DO THIS

✓ Use Environment Variables 
// .env.local (git ignored)

// Access in code
const key = process.env.STRIPE_SECRET_KEY

Cloud Security Checklist

□ Secrets in env vars

Never in code, never in git

□ HTTPS everywhere

Vercel/Supabase handle this automatically

□ Row Level Security

Enable RLS on all Supabase tables

□ Validate auth server-side

Don't trust client-only auth checks

□ Configure CORS properly

Restrict to your domains

□ Rate limiting

Protect APIs from abuse
